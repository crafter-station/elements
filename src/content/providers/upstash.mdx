## Overview

Self-contained components for visualizing Upstash data. Rate limits, counters, queues, leaderboards, and cache status. Works with any Upstash product.

## Components

### Rate Limit

Visual rate limit indicator with progress bar and reset timer.

<ComponentPreviewItem
  componentKey="upstash-ratelimit"
  installUrl="@elements/upstash-ratelimit"
  category="Integration"
  name="Upstash Ratelimit"
>
  <UpstashRatelimit />
</ComponentPreviewItem>

```tsx
import { UpstashRatelimit } from "@/components/elements/upstash/upstash-ratelimit";

export function RateLimitStatus({ result }) {
  return (
    <UpstashRatelimit
      limit={result.limit}
```

## Usage with @upstash/ratelimit

```tsx
import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";
import { UpstashRatelimit } from "@/components/elements/upstash/upstash-ratelimit";

const ratelimit = new Ratelimit({
  redis: Redis.fromEnv(),
  limiter: Ratelimit.slidingWindow(10, "10 s"),
});

export async function RateLimitedAPI() {
  const { success, limit, remaining, reset } = await ratelimit.limit("user_123");

  return (
    <UpstashRatelimit
      success={success}
      limit={limit}
      remaining={remaining}
      reset={reset}
    />
  );
}
```
