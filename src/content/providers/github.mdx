## Installation

Install any component individually:

```bash
bunx shadcn@latest add @elements/github-repo-card
bunx shadcn@latest add @elements/github-stars
bunx shadcn@latest add @elements/github-contributions
bunx shadcn@latest add @elements/github-languages
bunx shadcn@latest add @elements/github-star-button
bunx shadcn@latest add @elements/github-profile-card
bunx shadcn@latest add @elements/github-activity-calendar
```

## Environment Variables

For higher API rate limits, set a GitHub token:

```bash
GITHUB_TOKEN=your_github_token
```

Without a token, you're limited to 60 requests/hour. With a token, 5,000 requests/hour.

## Usage

```tsx

import { GitHubStars } from "@/components/elements/github/github-stars";
import { GitHubContributions } from "@/components/elements/github/github-contributions";
import { GitHubLanguages } from "@/components/elements/github/github-languages";

import { GitHubProfileCard } from "@/components/elements/github/github-profile-card";
import { GitHubActivityCalendar } from "@/components/elements/github/github-activity-calendar";

export function RepoMetrics() {
  return (
    <div className="space-y-4">
      <GitHubRepoCard owner="shadcn-ui" repo="ui" />
      <GitHubStars owner="shadcn-ui" repo="ui" />
      <GitHubContributions owner="shadcn-ui" repo="ui" />
      <GitHubLanguages owner="shadcn-ui" repo="ui" />
      <GitHubStarButton owner="shadcn-ui" repo="ui" />
      <GitHubProfileCard username="shadcn" />
      <GitHubActivityCalendar username="shadcn" />
    </div>
  );
}
```
